    [docs]    def __iter__(self):
        """
        Traverses the file

        :yield: next line in the file (or 'timed out' if it times-out)
        """
        line = None
        while line != EOF:
            try:
                line =  self.file.readline()
                yield line
            except socket.timeout:
                self.logger.debug('socket timed out')
                yield TIMED_OUT
        return
